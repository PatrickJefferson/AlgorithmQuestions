### 编程判断俩个链表是否相交
题目：给出俩个单向链表的头指针，比如h1，h2，判断这俩个链表是否相交。
为了简化问题，我们假设俩个链表均不带环。
问题扩展：
1.如果链表可能有环呢?
2.如果需要求出俩个链表相交的第一个节点呢?

#### 解析:

1. 链表相交:

   - 两单向链表若不带环则必为Y型

   ​	\-\-\-
    	   \|\-\-\-\-\- 
   ​	\-\-\-

   - 如果两单向链表都带环相交则环为同一环, 两链表的入口为同一节点或者一链表的入口在另一链表的环上
   - 如果两链表之一带环则不可能相交, 因为如果相交则环一定是共有的

2. 判断链表是否带环:

   - 差速法: 两指针同时从头节点出发, 速度分别为1和2, 开始遍历后如果两指针在某一节点相遇则说明该链表带环, 如果快指针遍历结束则说明链表无环

3. 链表带环获取入口:

   - 保存快慢指针第一次相遇的节点, 设置两指针分别从该节点和头节点同时以速度为1开始遍历, 当两指针第一次相遇则该节点为环的入口

4. 获取两链表交点:

   - 带环则考虑交点在环上的情况, 即两环入口不一样的情况, 此时只需从一环的入口处开始遍历到下次到达入口前是否存在节点为另一环的入口即可
   - 如果交点在为环的入口或者在入口前则与Y型链表交点一致, 获取两链表长度或者到入口处(两链表带环且入口一致)的长度, 并获取长度差, 将较长链表首先遍历到差值节点处再同时遍历两链表, 当两链表节点首次相同时则为入口

class Node():
    '''节点类'''
    def __init__(self, value, p=None):
        self.value = value
        self.next = p


class LinkList():
    '''链表类'''
    def __init__(self):
        self.head = None

    # 初始化链表
    def init_list(self, data):
        self.head = Node(data[0])
        p = self.head

        for item in data[1:]:
            p.next = Node(item)
            p = p.next

    # 获取链表长度
    def get_length(self):
        p = self.head
        length = 0
        while p!=None:
            length += 1
            if p.next == self.head:
                # 当出现环的时候则终止循环计数
                break
            p = p.next

        return length

    # 判断链表是否为空
    def is_empty(self):
        return self.get_length() == 0

    # 清除链表
    def clear(self):
        self.head = None

    # 追加节点
    def append(self, value):
        node = Node(value)
        if self.head:
            p = self.head
            while p.next:
                p = p.next
            p.next = node
        else:
            self.head = node

    # 获取指定序号的节点
    def get_node(self, index):
        if index >= self.get_length():
            raise IndexError('下标越界!')
        else:
            p = self.head
            for i in range(index):
                p = p.next
            return p

    # 插入节点
    def insert(self, index, value):
        node = Node(value)
        if index >= self.get_length():
            raise IndexError('下标越界!')
        else:
            if index == 0:
                node.next, self.head = self.head, node
            else:
                p = self.head
                for i in range(index - 1):
                    p = p.next
                p.next, node.next = node, p.next

    # 删除节点
    def delete(self, index):
        if index >= self.get_length():
            raise IndexError('下标越界!')
        else:
            if index == 0:
                self.head = self.head.next
            else:
                p = self.head
                for i in range(index - 1):
                    p = p.next
                p.next = p.next.next


class DoubleLink():

    def __init__(self, list1, list2):
        self.list1 = list1
        self.list2 = list2

    # 获取链表的环入口
    def have_loop(self):

        node1 = None
        node2 = None
        # 判断list1是否存在环 - 差速法
        p1 = self.list1.head
        p2 = self.list1.head
        while p2.next:
            p1 = p1.next
            p2 = p2.next
            if not p2.next:
                break
            p2 = p2.next
            if p1 == p2:
                node1 = p1
                break
        # 判断list2中是否有环 - 差速法
        p1 = self.list2.head
        p2 = self.list2.head
        while p2.next:
            p1 = p1.next
            p2 = p2.next
            if not p2.next:
                break
            p2 = p2.next
            if p1 == p2:
                node2 = p1
                break

        entrance1 = None
        entrance2 = None
        # 获取链表1的环入口
        if node1:
            p1 = node1
            p2 = self.list1.head
            while p1 != p2:
                p1 = p1.next
                p2 = p2.next
            entrance1 = p1
        # 获取链表2的环入口
        if node2:
            p1 = node2
            p2 = self.list2.head
            while p1 != p2:
                p1 = p1.next
                p2 = p2.next
            entrance2 = p1

        return entrance1, entrance2

    def intersect(self):
        # 初始化交点为None
        result = None
        entrance1, entrance2 = self.have_loop()
        if entrance1 and entrance2:
            # 如果都存在环
            if entrance1 == entrance2:
                # 如果入口相同则相交
                # 获取链表到入口处的长度
                length1 = 0
                length2 = 0
                p1 = self.list1.head
                p2 = self.list2.head
                while p1 != entrance1 or p2 != entrance2:
                    if p1 != entrance1:
                        p1 = p1.next
                        length1 += 1
                    if p2 != entrance2:
                        p2 = p2.next
                        length2 += 1
                p1 = self.list1.head
                p2 = self.list2.head
                sub = length1 - length2
                # 获取长度差并提前将较长的链表开始遍历差值长度
                if sub > 0:
                    for i in range(sub):
                        p1 = p1.next
                else:
                    for i in range(sub * (-1)):
                        p2 = p2.next
                while p1 != p2:
                    p1 = p1.next
                    p2 = p2.next
                result = p1

            else:
                # 如果入口不同则判断两个入口是否在同一个环上
                p1 = entrance1
                p2 = entrance1
                while p1.next != p2:
                    p1 = p1.next
                    if p1 == entrance2:
                        # 如果链表2的入口在链表1的环中则说明相交, 且交点为任一入口
                        return entrance1

        elif not entrance1 and not entrance2:
            # 如果都不存在环
            p1 = self.list1.head
            p2 = self.list2.head
            while p1.next:
                p1 = p1.next
            while p2.next:
                p2 = p2.next
            if p1 == p2:
                # 获取链表到入口处的长度
                length1 = self.list1.get_length()
                length2 = self.list2.get_length()
                p1 = self.list1.head
                p2 = self.list2.head
                sub = length1 - length2
                # 获取长度差并提前将较长的链表开始遍历差值长度
                if sub > 0:
                    for i in range(sub):
                        p1 = p1.next
                else:
                    for i in range(sub * (-1)):
                        p2 = p2.next
                while p1 != p2:
                    p1 = p1.next
                    p2 = p2.next
                result = p1

        return result


def intersect(list1, list2):
    p1 = list1.head
    p2 = list2.head
    while p1.next:
        p1 = p1.next
    while p2.next:
        p2 = p2.next
    if id(p1) == id(p2):
        return True
    else:
        return False


def main():
    # 初始化链表数据
    data = [0,1,2,3,4,5,6,7]
    data2 = [3,5,2,1]
    data3 = [8,3,4,7,4,1,3]
    link_list1 = LinkList()
    link_list1.init_list(data)
    link_list2 = LinkList()
    link_list2.init_list(data2)
    link_list3 = LinkList()
    link_list3.init_list(data3)
    # 不带环的情况
    node1 = link_list1.get_node(7)
    node2 = link_list2.get_node(3)
    node3 = link_list3.head
    print(node1.value, node2.value, node3.value)
    print('测试两无环不相交:')
    link_list_string = '''

            3-5-2-1
    '''
    print(link_list_string)
    # 创建相交问题对象
    doubleLink = DoubleLink(link_list1, link_list2)
    print('交点:', doubleLink.intersect())
    print('-'*20)
    # 创建无环相交链表
    link_list_string = '''
    0-1-2-3-4-5-6-7
                  |
                  8-3-4-7-4-1-3
                  |

    node1.next = node3
    node2.next = node3
    print('测试无环相交:')
    print(link_list_string)
    print('交点:', doubleLink.intersect().value)
    print('-' * 20)
    # 改变链表3为环链表
    link_list_string = '''

                  |/    |
                  8-3-4-7
                  |

    entrances = doubleLink.have_loop()
    print('两链表环入口:', entrances[0].value, entrances[1].value)
    print('交点:', doubleLink.intersect().value)
    print('-' * 20)
    # 改变链表3为有入口的环链表
    link_list_string = '''
    0-1-2-3-4-5-6-7   3-1-4
                  |   |  /
                  8-3-4-7
                  |
            3-5-2-1
    '''
    node31.next = link_list3.get_node(2)
    print('测试两链表在环入口前相交:')
    print(link_list_string)
    entrances = doubleLink.have_loop()
    print('两链表环入口:', entrances[0].value, entrances[1].value)
    print('交点:', doubleLink.intersect().value)
    print('-' * 20)
    # 改变链表相交在环上, 即有两个入口
    link_list_string = '''
                      3-1-4
                     /    |

                          |
                    3-5-2-1
    '''
    node31.next = link_list3.head
    node2.next = link_list3.get_node(3)
    print('测试两链表在环上相交, 即环有两个入口:')
    print(link_list_string)
    entrances = doubleLink.have_loop()
    print('两链表环入口:', entrances[0].value, entrances[1].value)
    print('交点:', doubleLink.intersect().value)
    print('-' * 20)


if __name__ == '__main__':
    main()
